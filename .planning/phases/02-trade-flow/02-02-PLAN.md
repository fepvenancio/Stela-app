```typescript
'use client'

interface BlendedRateSummaryProps {
  /** Individual APR scores from ranked trades. Null scores are excluded. */
  scores: (number | null)[]
  /** Mode determines label text */
  mode: 'lending' | 'swap'
}

export function BlendedRateSummary({ scores, mode }: BlendedRateSummaryProps) {
  // Filter to valid scores only
  const validScores = scores.filter((s): s is number => s !== null)

  // Only show when 2+ orders are selected (per UI-SPEC: "Only show when 2+ orders are selected for multi-fill")
  if (validScores.length < 2) return null

  // Simple average (all ranked trades are equally weighted in BestTradesPanel display context)
  const blended = validScores.reduce((sum, s) => sum + s, 0) / validScores.length

  const label = mode === 'swap' ? 'Blended Rate' : 'Blended APR'

  return (
    <div className="flex items-center justify-between px-3 py-2 rounded-lg bg-surface/20 border border-edge/20">
      <span className="text-[10px] uppercase tracking-widest font-bold text-dust">
        {label}
      </span>
      <span className="font-mono text-sm text-nebula">
        {blended.toFixed(2)}%
      </span>
    </div>
  )
}
```

```typescript
'use client'

import { useState } from 'react'
import { Loader2 } from 'lucide-react'
import { findTokenByAddress } from '@fepvenancio/stela-sdk'
import {
  Dialog,
  DialogContent,
  DialogHeader,
  DialogTitle,
} from '@/components/ui/dialog'
import { Button } from '@/components/ui/button'
import { Input } from '@/components/ui/input'
import { FeeBreakdown } from '@/components/FeeBreakdown'
import { Web3ActionWrapper } from '@/components/Web3ActionWrapper'
import { formatAddress } from '@/lib/address'

interface QuickLendModalProps {
  open: boolean
  onClose: () => void
  debtToken: string
  collateralToken: string
}

export function QuickLendModal({ open, onClose, debtToken, collateralToken }: QuickLendModalProps) {
  const [amount, setAmount] = useState('')
  const [isPending, setIsPending] = useState(false)

  const debtInfo = findTokenByAddress(debtToken)
  const collInfo = findTokenByAddress(collateralToken)
  const debtSymbol = debtInfo?.symbol ?? formatAddress(debtToken)
  const collSymbol = collInfo?.symbol ?? formatAddress(collateralToken)

  // Per UI-SPEC copywriting: modal title = "Lend {debtSymbol} / {collSymbol}"
  const title = `Lend ${debtSymbol} / ${collSymbol}`

  const handleSign = async () => {
    // Navigate to trade page with pre-filled params for actual signing flow
    // QuickLendModal provides the low-friction entry point; full signing uses the Trade page infrastructure
    setIsPending(true)
    try {
      window.location.href = `/trade?debtToken=${debtToken}&collateralToken=${collateralToken}&mode=lend&amount=${encodeURIComponent(amount)}`
    } finally {
      setIsPending(false)
    }
  }

  return (
    <Dialog open={open} onOpenChange={(v) => { if (!v) onClose() }}>
      <DialogContent className="max-w-md bg-abyss border-edge/30">
        <DialogHeader>
          <DialogTitle className="text-chalk font-display text-xs uppercase tracking-wider">
            {title}
          </DialogTitle>
        </DialogHeader>

        <div className="space-y-4 pt-2">
          {/* Amount input */}
          <div>
            <label className="text-[10px] uppercase tracking-widest font-bold text-dust mb-1.5 block">
              Amount
            </label>
            <Input
              type="text"
              inputMode="decimal"
              placeholder={`0.00 ${debtSymbol}`}
              value={amount}
              onChange={(e) => setAmount(e.target.value)}
              className="font-mono text-sm bg-surface/30 border-edge/30"
            />
          </div>

          {/* Fee breakdown — per UI-SPEC: must appear above the Sign button */}
          <FeeBreakdown type="lending" />

          {/* Sign button — gated by wallet connection */}
          <Web3ActionWrapper>
            <Button
              onClick={handleSign}
              disabled={!amount || isPending}
              className="w-full min-h-[36px] bg-star hover:bg-star/90 text-void font-medium"
            >
              {isPending ? (
                <>
                  <Loader2 className="h-4 w-4 animate-spin mr-2" />
                  Signing...
                </>
              ) : (
                'Sign Lend Offer'
              )}
            </Button>
          </Web3ActionWrapper>
        </div>
      </DialogContent>
    </Dialog>
  )
}
```

The QuickLendModal acts as a streamlined entry point that captures the user's intent (pair + amount) and navigates to the Trade page with pre-filled params for the full signing flow. This avoids duplicating the complex signing infrastructure while providing the quick-action UX of NAV-03. The Trade page (with nuqs from Plan 01) will read the params and pre-fill the form.
